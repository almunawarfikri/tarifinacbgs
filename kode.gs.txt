const SHEET_NAME = "DATA";

function doGet(e) {
  // Jika diakses lewat browser langsung atau Google Apps Script Web App
  if (!e || !e.parameter || !e.parameter.action) {
    return HtmlService.createHtmlOutputFromFile('index')
      .setTitle('Tarif INACBG')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  }

  // Jika diakses sebagai API (Misal: ?action=getData)
  const action = e.parameter.action;
  if (action === "getData") {
    const data = getData();
    return ContentService.createTextOutput(JSON.stringify(data))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  const params = JSON.parse(e.postData.contents);
  const action = params.action;

  if (action === "addData") {
    addData(params.data);
    return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT);
  }

  if (action === "updateData") {
    updateData(params.rowIndex, params.data);
    return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT);
  }
}

function getData() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  return sheet.getDataRange().getDisplayValues();
}

function addData(data) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  sheet.appendRow([
    data.diagnosa,
    data.kode,
    data.kelas1,
    data.kelas2,
    data.kelas3,
    data.severity,
    data.syarat,
    data.cluster
  ]);
}

function updateData(rowIndex, data) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  sheet.getRange(rowIndex, 1, 1, 8).setValues([[
    data.diagnosa,
    data.kode,
    data.kelas1,
    data.kelas2,
    data.kelas3,
    data.severity,
    data.syarat,
    data.cluster
  ]]);
}
