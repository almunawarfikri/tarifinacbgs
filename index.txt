<script>
let globalData = [];
let editRow = null;
let currentSort = { col: null, dir: 'asc' };

function loadData() {
  google.script.run.withSuccessHandler(data => {
    const header = data[0];
    const topupIdx = header.findIndex(h => h.toLowerCase().replace(/\s/g, '').includes("topup"));
    
    globalData = data.map(row => {
      if (topupIdx !== -1) return row.filter((_, idx) => idx !== topupIdx);
      return row;
    });
    renderTable(globalData);
  }).getData();
}

function renderTable(data) {
  let html = "<thead><tr>";
  if (data.length > 0) {
    data[0].forEach((col, j) => {
      let sortClass = "";
      if (currentSort.col === j) sortClass = currentSort.dir === 'asc' ? 'sort-asc' : 'sort-desc';
      html += `<th class="${sortClass}" onclick="sortTable(${j})">${col}</th>`;
    });
    html += "<th>EDIT</th></tr></thead><tbody>";

    data.slice(1).forEach((row) => {
      const originalIndex = globalData.findIndex(r => r === row);
      html += "<tr>";
      row.forEach(col => html += `<td>${col}</td>`);
      html += `<td>
        <div class="action-cell">
          <button class="btn-edit" onclick="editData(${originalIndex})">✏️</button>
        </div>
      </td></tr>`;
    });
  }
  html += "</tbody>";
  document.getElementById("table").innerHTML = html;
}

function sortTable(n) {
  const dir = (currentSort.col === n && currentSort.dir === 'asc') ? 'desc' : 'asc';
  currentSort = { col: n, dir: dir };
  const header = globalData[0];
  const rows = globalData.slice(1);

  rows.sort((a, b) => {
    let x = (a[n] || "").toString().toLowerCase();
    let y = (b[n] || "").toString().toLowerCase();
    const cleanNum = s => parseFloat(s.replace(/\./g, '').replace(/,/g, '.'));
    if (!isNaN(cleanNum(x)) && !isNaN(cleanNum(y))) {
       x = cleanNum(x); y = cleanNum(y);
    }
    if (x < y) return dir === 'asc' ? -1 : 1;
    if (x > y) return dir === 'asc' ? 1 : -1;
    return 0;
  });
  globalData = [header, ...rows];
  renderTable(globalData);
}

function filterData() {
  const key = document.getElementById("search").value.toLowerCase();
  const filtered = [globalData[0], ...globalData.slice(1).filter(row => row.join(" ").toLowerCase().includes(key))];
  renderTable(filtered);
}

function openAdd() {
  editRow = null;
  document.getElementById("modalTitle").innerText = "Tambah Data";
  document.querySelectorAll("#modal input").forEach(i => i.value = "");
  showModal();
}

function editData(index) {
  editRow = index;
  const row = globalData[index];
  const fields = ["diagnosa", "kode", "kelas1", "kelas2", "kelas3", "severity", "syarat", "cluster"];
  fields.forEach((f, i) => document.getElementById(f).value = row[i] || "");
  document.getElementById("modalTitle").innerText = "Edit Data";
  showModal();
}

function save() {
  const data = {
    diagnosa: document.getElementById("diagnosa").value,
    kode: document.getElementById("kode").value,
    kelas1: document.getElementById("kelas1").value,
    kelas2: document.getElementById("kelas2").value,
    kelas3: document.getElementById("kelas3").value,
    severity: document.getElementById("severity").value,
    syarat: document.getElementById("syarat").value,
    cluster: document.getElementById("cluster").value
  };
  
  if (editRow) {
    google.script.run.withSuccessHandler(loadData).updateData(editRow + 1, data);
  } else {
    google.script.run.withSuccessHandler(loadData).addData(data);
  }
  closeModal();
}

function showModal() {
  document.getElementById("modal").style.display = "block";
  document.getElementById("overlay").style.display = "block";
}

function closeModal() {
  document.getElementById("modal").style.display = "none";
  document.getElementById("overlay").style.display = "none";
}

loadData();
</script>
